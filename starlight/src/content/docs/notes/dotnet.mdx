## Convert to Central Time

```csharp
public static DateTime ToCentralTime(this string dateTimeString)
{
    TimeZoneInfo centralTimeZone = TimeZoneInfo.FindSystemTimeZoneById("Central Standard Time");
    return DateTime.TryParse(dateTimeString, out DateTime dt) ? TimeZoneInfo.ConvertTimeFromUtc(dt, centralTimeZone) : DateTime.MinValue;
}

public static DateTime ToCentralTime(this DateTimeOffset dateTime)
{
    TimeZoneInfo centralTimeZone = TimeZoneInfo.FindSystemTimeZoneById("Central Standard Time");
    return TimeZoneInfo.ConvertTime(dateTime, centralTimeZone).DateTime;
}
```

## Install .NET SDK

### Linux

#### Install .NET SDK 10

- Fedora

  ```bash frame="none"
  sudo dnf install dotnet-sdk-10.0
  sudo dotnet workload update
  ```

- Debian / Ubuntu

  ```bash frame="none"
  sudo apt install dotnet-sdk-10.0
  sudo dotnet workload update
  ```

#### Install .NET SDK 11

##### Installation Scripts

- Create the `remove-dotnet-11.sh` and `update-dotnet-11.sh` script files

  <details>
      <summary>remove-dotnet-11.sh</summary>
  
      ```bash
      #!/bin/bash
  
      install_dir=
  
      read -ra dirs <<<$(whereis dotnet)
  
      for dir in "${dirs[@]:1}"; do
          if [ -d "${dir}/sdk" ]; then
              install_dir="${dir}"
          fi
      done
  
      if [[ $install_dir ]]; then
          echo "Removing old versions of .NET 11 in ${install_dir}"
  
          dir_pattern=11*
  
          sudo rm -rf $install_dir/host/fxr/$dir_pattern
          sudo rm -rf $install_dir/packs/*/$dir_pattern
          sudo rm -rf $install_dir/sdk/$dir_pattern
          sudo rm -rf $install_dir/sdk-manifests/$dir_pattern
          sudo rm -rf $install_dir/shared/*/$dir_pattern
          sudo rm -rf $install_dir/templates/$dir_pattern
      fi
      ```
  </details>
  
  <details>
      <summary>update-dotnet-11.sh</summary>
  
      ```bash {14-15} {17-18} {20-21}
      #!/bin/bash
  
      install_dir=
  
      read -ra dirs <<<$(whereis dotnet)
  
      for dir in "${dirs[@]:1}"; do
              if [ -d "${dir}/sdk" ]; then
                      install_dir="${dir}"
              fi
      done
  
      if [[ -z $install_dir ]]; then
          # Arch Linux / Debian
          install_dir=/usr/share/dotnet
  
          # Fedora
          install_dir=/usr/lib64/dotnet/sdk
  
          # Ubuntu
          install_dir=/usr/lib/dotnet/sdk
      fi
  
      if [[ $install_dir ]]; then
              echo -e "\nInstalling gpg"
              if [[ ! -z $(which apt 2> /dev/null) ]]; then
                      sudo apt install gpg
              elif [[ ! -z $(which dnf 2> /dev/null) ]]; then
                      sudo dnf install gpg
              fi
  
              echo -e "\nDownloading latest dotnet-install.asc"
              wget https://dot.net/v1/dotnet-install.asc -O dotnet-install.asc
  
              echo -e "\nImporting signature into gpg"
              gpg --import dotnet-install.asc
  
              echo -e "\nDownloading latest dotnet-install.sh"
              wget https://dot.net/v1/dotnet-install.sh -O dotnet-install.sh
              wget https://dot.net/v1/dotnet-install.sig -O dotnet-install.sig
              gpg --verify dotnet-install.sig dotnet-install.sh
              chmod u+x dotnet-install.sh
  
              ./remove-dotnet-11.sh
  
              echo -e "\nInstalling latest version of .NET 11 in ${install_dir}"
              sudo ./dotnet-install.sh --channel 11.0 --install-dir $install_dir --no-path
      fi
  
      echo "Done."
      ```
  </details>

- Allow the execution of these script files

  ```bash frame="none"
  chmod +x remove-dotnet-11.sh
  chmod +x update-dotnet-11.sh
  ```

##### Installing/Updating

- Install/Update .NET SDK 11 (pre-release version) using the installation script files

  ```bash frame="none"
  ./update-dotnet-11.sh
  sudo dotnet workload update
  ```
